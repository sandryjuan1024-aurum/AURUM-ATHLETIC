    <script>
        document.getElementById('form-registro').addEventListener('submit', function(event) {
            // Evita que la página se recargue al enviar el formulario
            event.preventDefault();

            // Capturar los valores de los campos
            const nombre = document.getElementById('nombre').value.trim();
            const correo = document.getElementById('correo').value.trim();
            const interes = document.getElementById('interes').value;
            const feedback = document.getElementById('mensaje-feedback');

            // Resetear el estado del cuadro de alerta
            feedback.style.display = 'none';
            feedback.className = 'mensaje-alerta';

            // Validación simple de campos vacíos
            if (nombre === '' || correo === '' || interes === '') {
                feedback.textContent = '❌ Por favor, completa todos los campos del formulario antes de continuar.';
                feedback.classList.add('alerta-error');
                feedback.style.display = 'block';
                return;
            }

            // Si pasa la validación, muestra el mensaje de éxito dinámico
            feedback.innerHTML = `🏆 ¡Bienvenido al Club, <strong>${nombre}</strong>!<br>Hemos registrado tu meta de <strong>${interes}</strong>. Tu cupón de descuento del 10% ha sido enviado a <em>${correo}</em>.`;
            feedback.classList.add('alerta-exito');
            feedback.style.display = 'block';

            // Limpiar los campos del formulario para un nuevo registro
            document.getElementById('form-registro').reset();
        });
    </script>
